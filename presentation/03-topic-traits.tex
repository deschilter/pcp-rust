    fn get_val() -> String {
        String::from("foo")
    }
}\end{lstlisting}

    \pause
    Trait implementation:
    \begin{lstlisting}[language=Rust,escapechar=@,label={lst:traits2}]

impl Foo for Bar {
    fn get_val() -> String {
        String::from("bar")
    }
}\end{lstlisting}
